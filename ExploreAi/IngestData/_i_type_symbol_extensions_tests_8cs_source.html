448 private async Task<(ITypeSymbol, Compilation)> GetParameterType(string code)
449 {
450 var tree = CSharpSyntaxTree.ParseText(code);
451
452 var compilation = CSharpCompilation.Create(Guid.NewGuid().ToString("N"),
453 syntaxTrees: new[] { tree },
454 references: AssemblyReferences.GetMetadataReferences(new[]
455 {
456 typeof(object).Assembly,
457 typeof(BusinessBase<>).Assembly
458 }));
459
460 var root = await tree.GetRootAsync().ConfigureAwait(false);
461 var model = compilation.GetSemanticModel(tree);
462 var methodSymbol = model.GetDeclaredSymbol(
463 root.DescendantNodes().OfType<MethodDeclarationSyntax>().First(_ => _.Identifier.Text == "Foo"));
464 return (methodSymbol.Parameters[0].Type, compilation);
465 }

467 private async Task<(SyntaxNode, SemanticModel)> GetRootAndModel(string code)
468 {
469 var tree = CSharpSyntaxTree.ParseText(code);
470
471 var compilation = CSharpCompilation.Create(Guid.NewGuid().ToString("N"),
472 syntaxTrees: new[] { tree },
473 references: AssemblyReferences.GetMetadataReferences(new[]
474 {
475 typeof(object).Assembly,
476 typeof(BusinessBase<>).Assembly
477 }));
478
479 var model = compilation.GetSemanticModel(tree);
480 var root = await tree.GetRootAsync().ConfigureAwait(false);
481
482 return (root, model);
