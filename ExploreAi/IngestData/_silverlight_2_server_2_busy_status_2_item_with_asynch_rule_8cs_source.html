88 public FiveSecondsLongRule(Csla.Core.IPropertyInfo primaryProperty)
89 : base(primaryProperty)

91 IsAsync = true;

94
95 protected override void Execute(IRuleContext context)
96 {
97 if (Csla.ApplicationContext.LogicalExecutionLocation == Csla.ApplicationContext.LogicalExecutionLocations.Client)
98 {
99 BackgroundWorker worker = new BackgroundWorker();
100
101 worker.DoWork += (s, e) =>
102 {
103 System.Threading.Thread.Sleep(2000);
104 var value = context.InputPropertyValues[PrimaryProperty];
105 if (value == null || value.ToString().ToUpper() == "ERROR")
106 context.AddErrorResult("error detected");
107 };
108
109 worker.RunWorkerCompleted += (s, e) => context.Complete();
110
111 // simulating an asynchronous process.
112 worker.RunWorkerAsync();
113 }
114 else
115 {
116 var value = context.InputPropertyValues[PrimaryProperty];
117 if (value == null || value.ToString().ToUpper() == "ERROR")
118 context.AddErrorResult("error detected");
119 context.Complete();
120 }
121 }
122 }
