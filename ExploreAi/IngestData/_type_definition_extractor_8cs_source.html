31 public static ExtractedTypeDefinition ExtractTypeDefinition(DefinitionExtractionContext extractionContext, TypeDeclarationSyntax targetTypeDeclaration)
32 {
33 ExtractedTypeDefinition definition = new ExtractedTypeDefinition();
34 StringBuilder fullyQualifiedNameBuilder = new StringBuilder();
35
36 definition.TypeName = GetTypeName(extractionContext, targetTypeDeclaration);
37 definition.TypeKind = GetTypeKind(extractionContext, targetTypeDeclaration);
38 definition.Namespace = GetNamespace(extractionContext, targetTypeDeclaration);
39 definition.Scope = GetScopeDefinition(extractionContext, targetTypeDeclaration);
40
41 foreach (ExtractedContainerDefinition containerDefinition in ContainerDefinitionsExtractor.GetContainerDefinitions(extractionContext, targetTypeDeclaration))
42 {
43 definition.ContainerDefinitions.Add(containerDefinition);
44 fullyQualifiedNameBuilder.Append(containerDefinition.Name);
45 fullyQualifiedNameBuilder.Append(".");
46 }
47
48 foreach (ExtractedPropertyDefinition propertyDefinition in PropertyDefinitionsExtractor.ExtractPropertyDefinitions(extractionContext, targetTypeDeclaration))
49 {
50 definition.Properties.Add(propertyDefinition);
51 }
52
53 foreach (ExtractedFieldDefinition fieldDefinition in FieldDefinitionsExtractor.ExtractFieldDefinitions(extractionContext, targetTypeDeclaration))
54 {
55 definition.Fields.Add(fieldDefinition);
56 }
57
58 fullyQualifiedNameBuilder.Append(definition.TypeName);
59 definition.FullyQualifiedName = fullyQualifiedNameBuilder.ToString();
60
61 return definition;
62 }

72 private static string GetNamespace(DefinitionExtractionContext extractionContext, TypeDeclarationSyntax targetTypeDeclaration)
73 {
74 string namespaceName = string.Empty;
75 NamespaceDeclarationSyntax namespaceDeclaration;
76 TypeDeclarationSyntax containingTypeDeclaration;
77
78 // Iterate through the containing types should the target type be nested inside other types
79 containingTypeDeclaration = targetTypeDeclaration;
80 while (containingTypeDeclaration.Parent is TypeDeclarationSyntax)
81 {
82 containingTypeDeclaration = (TypeDeclarationSyntax)containingTypeDeclaration.Parent;
83 }
84
85 namespaceDeclaration = containingTypeDeclaration.Parent as NamespaceDeclarationSyntax;
86 if (namespaceDeclaration is not null)
87 {
88 namespaceName = namespaceDeclaration.Name.ToString();
89 }
90
91 return namespaceName;
92 }

100 private static string GetScopeDefinition(DefinitionExtractionContext extractionContext, TypeDeclarationSyntax targetTypeDeclaration)
101 {
102 StringBuilder scopeNameBuilder = new StringBuilder();
103
104 foreach (SyntaxToken modifier in targetTypeDeclaration.Modifiers)
105 {
106 if (modifier.IsKind(Microsoft.CodeAnalysis.CSharp.SyntaxKind.PublicKeyword))
107 {
108 AppendScopeName(scopeNameBuilder, modifier.ValueText);
109 continue;
110 }
111 if (modifier.IsKind(Microsoft.CodeAnalysis.CSharp.SyntaxKind.InternalKeyword))
112 {
113 AppendScopeName(scopeNameBuilder, modifier.ValueText);
114 continue;
115 }
116 if (modifier.IsKind(Microsoft.CodeAnalysis.CSharp.SyntaxKind.ProtectedKeyword))
117 {
118 AppendScopeName(scopeNameBuilder, modifier.ValueText);
119 continue;
120 }
121 if (modifier.IsKind(Microsoft.CodeAnalysis.CSharp.SyntaxKind.PrivateKeyword))
122 {
123 AppendScopeName(scopeNameBuilder, modifier.ValueText);
124 continue;
125 }
126 }
127
128 if (scopeNameBuilder.Length < 1)
129 {
130 scopeNameBuilder.Append("internal");
131 }
132
133 return scopeNameBuilder.ToString().Trim();
134 }
